-- Script: StaminaSystem
-- Ubicación: StarterGui > ScreenGui > LocalScript
-- Requiere en la GUI:
--   ScreenGui > Frame (fondo, llamado "FondoStamina") > Frame (barra verde, llamado "BarraStamina")

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoidRootPart = character:WaitForChild("HumanoidRootPart")
local humanoid = character:WaitForChild("Humanoid")

local gui = player.PlayerGui:WaitForChild("ScreenGui")
local barraStamina = gui:WaitForChild("FondoStamina"):WaitForChild("BarraStamina")

-- =============================================
-- CONFIGURACIÓN DE STAMINA
-- =============================================
local staminaMaxima = 100
local staminaActual = 100
local velocidadBajada = 15       -- Stamina que se pierde por segundo al correr normal
local velocidadSubida = 10       -- Stamina que se recupera por segundo al estar quieto
local velocidadSubidaLenta = 4   -- Stamina que se recupera por segundo al caminar agotado
local umbralMovimiento = 1

local velocidadNormal = 16
local velocidadAgotado = 8

local estaAgotado = false

-- =============================================
-- LOOP PRINCIPAL
-- =============================================
RunService.Heartbeat:Connect(function(deltaTime)
	local velocidad = humanoidRootPart.AssemblyLinearVelocity.Magnitude

	if velocidad > umbralMovimiento then
		if not estaAgotado then
			-- Corriendo con stamina: baja normal
			staminaActual = staminaActual - (velocidadBajada * deltaTime)
		else
			-- Caminando agotado: recupera lento
			staminaActual = staminaActual + (velocidadSubidaLenta * deltaTime)
		end
	else
		-- Quieto: recupera a velocidad normal
		staminaActual = staminaActual + (velocidadSubida * deltaTime)
	end

	staminaActual = math.clamp(staminaActual, 0, staminaMaxima)

	-- Actualizar barra visual
	local proporcion = staminaActual / staminaMaxima
	barraStamina.Size = UDim2.new(proporcion, 0, 1, 0)

	-- Cambiar velocidad según stamina
	if staminaActual <= 0 and not estaAgotado then
		humanoid.WalkSpeed = velocidadAgotado
		estaAgotado = true
	elseif staminaActual >= 20 and estaAgotado then
		humanoid.WalkSpeed = velocidadNormal
		estaAgotado = false
	end
end)
